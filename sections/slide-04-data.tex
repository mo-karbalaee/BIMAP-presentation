\begin{column}{0.56\textwidth}
{\small\textbf{Neurofinder (labeled, mouse)} \cite{neurofinder}}
{\footnotesize
\begin{itemize}\itemsep=0.2em
\item Labeled data, the scarce and expensive resource
\item 28 movies from 4 labs (19 train / 9 test)
\item Diverse brain regions, microscopes, indicators, SNR $\to$ generalization
\end{itemize}}
\medskip

{\small\textbf{Limitations}}
{\footnotesize
\begin{itemize}\itemsep=0.2em
\item Static per-recording footprints, not per-frame labels
\item $\Rightarrow$ a single frame hides inactive neurons; must aggregate over time
\item Label definition varies by lab: hand-annotated vs anatomical marker
\end{itemize}}
\end{column}
