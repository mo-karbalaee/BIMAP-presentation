\section{Critique}

\begin{frame}[t]{What we'd criticize in our own results}
\begin{columns}[t]
% ---- Statistics ----------------------------------------------------------
\begin{column}{0.5\textwidth}
{\footnotesize\color{black}%
\textbf{Our statistics don't prove what they seem to}\par\vspace{0.35em}
\begin{itemize}\itemsep=0.25em
\item The JEPA vs supervised gap is statistically significant, yet \textbf{tiny
      ($\Delta \approx 0.02$) and flips} with label fraction and resolution
      $\to$ significance here is misleading, not a real effect
\item For H3, $p$-values are meaningless at millions of pairs $\to$ only
      \textbf{effect size} is honest (Cohen's $d$: SSL 0.66, supervised 1.45,
      random 0.14)
\item Every result is a \textbf{single training seed} $\to$ we cannot separate a
      real effect from training noise
\end{itemize}}
\end{column}
\hfill%
% ---- What results do NOT show --------------------------------------------
\begin{column}{0.5\textwidth}
{\footnotesize\color{black}%
\textbf{What our results do NOT show}\par\vspace{0.35em}
\begin{itemize}\itemsep=0.25em
\item \textbf{Dice and mIoU are redundant} --- pixel overlap can look strong
      while neuron \textit{detection} stays poor (detection F1 $\sim$0.2 for all)
\item Our win over Cellpose is \textbf{not apples-to-apples} --- we are trained
      on the target, Cellpose is zero-shot and untuned
\item We \textbf{train and test on the same recording} $\to$ cross-recording
      generalization is unproven
\item Our SSL pretraining \textbf{never converges} (validation loss diverges)
      $\to$ these are floor numbers, not the method's real potential
\end{itemize}}
\end{column}
\end{columns}
\end{frame}
