\begin{columns}[t]
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Not a script: one full \textbf{pipeline}, single entry point
\item Two modes: \textbf{train} new models, \textbf{infer} on new recordings
\item Modular, swappable stages; segmenter = our \textbf{JEPA} model or a
      \textbf{Cellpose} baseline \cite{stringer2021cellpose}
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Every run logged (Dice, mIoU); checkpoints versioned, never overwritten
\item Reproducible: one config file, no hardcoded paths, clone and run
\item Meets and exceeds the Phase 0 MVP (load $\to$ segment $\to$ activity traces)
\end{itemize}
\end{column}
\end{columns}
\end{frame}
