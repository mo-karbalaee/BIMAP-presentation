\section{H1: Label efficiency}

\begin{frame}[t]{H1 --- Does self-supervision help when labels are scarce?}
\begin{columns}[c]
% ---- Left: terse bullets -------------------------------------------------
\begin{column}{0.50\textwidth}
{\small
\begin{itemize}\itemsep=0.2em
\item \textbf{Hypothesis:} self-supervised pretraining should help, most of all
      when labels are scarce
\item \textbf{Pretrain:} self-supervised (JEPA) on the calcium video with labels
      removed
\item \textbf{Fine-tune + test:} hold out one labeled recording, split it 80/20
      $\to$ fine-tune at 10--100\% of the train, test on the fixed 20\%
\item \textbf{Baseline:} from-scratch (same split)
\item \textbf{Metric:} Dice (held-out mouse) across label fractions
\item \textbf{Result:} JEPA-pretrained and from-scratch perform comparably across
      all label fractions
\end{itemize}}
\end{column}
