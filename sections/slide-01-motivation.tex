\section{Motivation}

\begin{frame}[t]{Calcium imaging: recording neuronal activity}
\begin{columns}[c]
% ---- Left: terse bullets -------------------------------------------------
\begin{column}{0.56\textwidth}
\begin{itemize}
\item Calcium imaging: a direct readout of brain activity
      \cite{grienberger2012calcium}
\item Neuron fires $\to$ Ca$^{2+}$ rises $\to$ indicator brightens $\to$ active
      neurons \enquote{light up} (2D${}+{}$time) \cite{chen2013gcamp6}
\item Records large populations of neurons in living, behaving animals
\item Raw movie $\neq$ science: each neuron must be \textbf{segmented} first
\item Manual annotation: slow, subjective, doesn't scale
\item \textbf{Goal:} automatic segmentation $\to$ fast, objective, scalable
\end{itemize}
\end{column}
